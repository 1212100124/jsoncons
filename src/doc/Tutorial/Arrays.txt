    #include "jsoncons/json.hpp"

    using jsoncons::json;
    using jsoncons::pretty_print;

## Constructing a one dimensional array

    json a = json::make_array(10,0);
	a[1] = 1;
	a[2] = 2;
    std::cout << pretty_print(a) << std::endl;

The output is

    [0,1,2,0,0,0,0,0,0,0]

## Constructing a two dimensional array

    json a = json::make_2d_array(10,8,0);
    a[0][0] = "Tenor";
    a[0][1] = "ATM vol";
	a[0][2] = "25-d-MS";
    a[0][3] = "25-d-RR";
    a[1][0] = "1Y";
    a[1][1] = 0.20;
    a[1][2] = 0.009;
    a[1][3] = -0.006;
    a[2][0] = "2Y";
    a[2][1] = 0.18;
    a[2][2] = 0.009;
    a[2][3] = -0.005;

    output_format format(true);
    format.precision(6);
    std::cout << pretty_print(a,format) << std::endl;

The output is

    [
        ["Tenor","ATM vol","25-d-MS","25-d-RR"],
        ["1Y",0.2,0.009,-0.006],
        ["2Y",0.18,0.009,-0.005]
    ]

## Constructing a three dimensional array
